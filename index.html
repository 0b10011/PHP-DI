Dependency Injection

An example is worth a thousand words:

use DI\Annotations\Inject;

class MyClass {
    /**
     * @Inject
     * @var MyService
     */
    private $service;

    public function doSomething($id) {
        $object = $service->getSomething($id);
    }
}
